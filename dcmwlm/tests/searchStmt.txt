# select-statement to get all PAs depending on the following variables
# :MODALITY
# :AETITLE
# :STARTDATE      -- Format: YYYYMMDD
# :STOPDATE       -- Format: YYYYMMDD
# :STARTTIME      -- Format: HH24MISS
# :STOPTIME       -- Format: HH24MISS
# :PHYSICIAN      -- name of performing physician
# :PATIENT_MIN    -- patient name lower bound
# :PATIENT_MIN    -- patient name upper bound
# :PATIENTID_MIN  -- patientID no lower bound
# :PATIENTID_MAX  -- patientID no upper bound
select k_hk.unr from k_hk, k_hk_basis_daten where k_hk.unr = k_hk_basis_daten.unr and hknr not like '%X%' and to_char(datum, 'YYYYMMDD') >= ':STARTDATE' and to_char(datum, 'YYYYMMDD') <= ':STOPDATE' and decode(labor, '1', 'VISUB', labor) = rtrim(':AETITLE', ' ')
